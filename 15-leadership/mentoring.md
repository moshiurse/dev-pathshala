## 📌 সিনিয়র ইঞ্জিনিয়ার মেন্টর হিসেবে

একজন সিনিয়র ইঞ্জিনিয়ারের প্রধান দায়িত্ব শুধু ভালো কোড লেখা নয়। আসল দায়িত্ব হলো
পুরো টিমকে লেভেল আপ করা। ধরুন আপনি Grameenphone এর ব্যাকএন্ড টিমে লিড —
আপনি একা যতই ভালো কোড লেখেন, যদি বাকি ৮ জন ডেভেলপার সেই মানে কোড না
লিখতে পারে, তাহলে প্রোডাক্ট কখনোই ভালো হবে না।

### 🏠 বাস্তব উদাহরণ: bKash এ নতুন জুনিয়র ডেভেলপার অনবোর্ডিং

## 🎯 পেয়ার প্রোগ্রামিং কৌশল

ব্যাকএন্ড টিমে নতুন ডেভেলপারদের দ্রুত অনবোর্ড করতে পেয়ার প্রোগ্রামিং ব্যাপকভাবে

### 🏠 PHP উদাহরণ: পেয়ার প্রোগ্রামিং সেশন — bKash Payment Validation

**সেশনের আগে (জুনিয়রের কোড):**

```php
<?php
// ❌ খারাপ কোড — জুনিয়র ডেভেলপারের প্রথম প্রচেষ্টা
// bKash পেমেন্ট ভ্যালিডেশন

function check_payment($data) {
    // কোনো টাইপ হিন্টিং নেই
    // ভ্যালিডেশন লজিক এলোমেলো
    // এরর হ্যান্ডলিং নেই

    if ($data['amount'] > 0) {
        if ($data['phone'] != '') {
            if (strlen($data['phone']) == 11) {
                if ($data['pin'] != '') {
                    // ডাটাবেসে সরাসরি কুয়েরি
                    $conn = mysqli_connect("localhost", "root", "", "bkash");
                    $sql = "SELECT * FROM users WHERE phone='" . $data['phone'] . "'";
                    $result = mysqli_query($conn, $sql);

                    if (mysqli_num_rows($result) > 0) {
                        $user = mysqli_fetch_assoc($result);
                        if ($user['balance'] >= $data['amount']) {
                            // পেমেন্ট প্রসেস
                            $new_balance = $user['balance'] - $data['amount'];
                            $sql2 = "UPDATE users SET balance=" . $new_balance
                                     . " WHERE phone='" . $data['phone'] . "'";
                            mysqli_query($conn, $sql2);
                            return "success";
                        } else {
                            return "no balance";
                        }
                    } else {
                        return "user not found";
                    }
                }
            }
        }
    }
    return "error";
```

**সেশনের পরে (মেন্টরের সাথে রিফ্যাক্টর করার পর):**

```php
<?php
// ✅ ভালো কোড — পেয়ার প্রোগ্রামিং সেশনের পরে
// bKash পেমেন্ট ভ্যালিডেশন — রিফ্যাক্টরড

declare(strict_types=1);

class PaymentValidationException extends RuntimeException {}
class InsufficientBalanceException extends PaymentValidationException {}
class UserNotFoundException extends PaymentValidationException {}

class PaymentRequest
{
    public function __construct(
        public readonly string $phone,
        public readonly float $amount,
        public readonly string $pin
    ) {
        $this->validate();
    }

    private function validate(): void
    {
        if ($this->amount <= 0) {
            throw new PaymentValidationException('পরিমাণ শূন্যের বেশি হতে হবে');
        }

        if (!preg_match('/^01[3-9]\d{8}$/', $this->phone)) {
            throw new PaymentValidationException('সঠিক বাংলাদেশি ফোন নম্বর দিন');
        }

        if (strlen($this->pin) < 4) {
            throw new PaymentValidationException('পিন কমপক্ষে ৪ সংখ্যার হতে হবে');
        }
    }
}

class PaymentService
{
    public function __construct(
        private readonly UserRepository $userRepo,
        private readonly TransactionLogger $logger
    ) {}

    public function processPayment(PaymentRequest $request): PaymentResult
    {
        $user = $this->userRepo->findByPhone($request->phone);

        if ($user === null) {
            throw new UserNotFoundException("ব্যবহারকারী পাওয়া যায়নি: {$request->phone}");
        }

        if ($user->getBalance() < $request->amount) {
            throw new InsufficientBalanceException('অপর্যাপ্ত ব্যালেন্স');
        }

        $transaction = $user->debit($request->amount);
        $this->userRepo->save($user);
        $this->logger->log($transaction);

        return new PaymentResult(
            success: true,
            transactionId: $transaction->getId(),
            remainingBalance: $user->getBalance()
        );
    }
}
```

**পরে (পেয়ার প্রোগ্রামিং সেশনের পরে):**

```javascript
// ✅ ভালো কোড — async/await, Strategy প্যাটার্ন, পরিষ্কার গঠন

const VEHICLE_CONFIG = {
    bike: { baseFare: 25, perKm: 2.0, trafficMultiplier: 1.5 },
    car:  { baseFare: 60, perKm: 5.0, trafficMultiplier: 1.3 },
    cng:  { baseFare: 40, perKm: 3.0, trafficMultiplier: 1.2 },
};

class FareCalculator {
    constructor(distanceService, trafficService, surgeService) {
        this.distanceService = distanceService;
        this.trafficService = trafficService;
        this.surgeService = surgeService;
    }

    async calculateFare(from, to, vehicleType) {
        const config = VEHICLE_CONFIG[vehicleType];
        if (!config) {
            throw new Error(`অজানা যানবাহনের ধরন: ${vehicleType}`);
        }

        const [distance, traffic, surgeFactor] = await Promise.all([
            this.distanceService.getDistance(from, to),
            this.trafficService.getTraffic(from, to),
            this.surgeService.getCurrentSurge(),
        ]);

        let fare = config.baseFare + (distance * config.perKm * surgeFactor);

        if (traffic === 'high') {
            fare *= config.trafficMultiplier;
        }

        return {
            fare: Math.round(fare),
            distance,
            vehicleType,
            surgeApplied: surgeFactor > 1.0,
        };
    }
}
```
